public static byte[] parse(String s)
{
    byte[] res = new byte[s.length()/2];
    int c = 0;
    for(int i =0; i&lt;s.length()-1; i++)
    {
        if(isHex(s.charAt(i)) &amp;&amp; isHex(s.charAt(i+1)))
        {
            res[c++] = parseByte(s.substring(i));
            i++;
        }
        else if ((s.charAt(i) == '?') &amp;&amp; (s.charAt(i+1) == '?')) {
            res[c++] = (byte) new Random().nextInt();
            i++;
        }
    }

    byte[] res2 = new byte[c];
    System.arraycopy(res, 0, res2, 0, c);
    return res2;
}

public static boolean isHex(char c) {
    return (c&gt;='0' &amp;&amp; c &lt;= '9') || (c&gt;='a' &amp;&amp; c &lt;= 'f')|| (c&gt;='A' &amp;&amp; c &lt;= 'F');
}


//转换字符串到byte

public static byte parseByte(String s) {
    if(s == null || s.isEmpty()) return 0;
    if(s.length() == 1)
        return parseChar(s.charAt(0));

    return (byte)(parseChar(s.charAt(0)) * 0x10 + parseChar(s.charAt(1)));
}

public static byte parseChar(char c) {
    if(c&gt;='0' &amp;&amp; c &lt;= '9') return (byte)(c - '0');
    if(c&gt;='a' &amp;&amp; c &lt;= 'f') return (byte)(c-'a'+10);
    if(c&gt;='A' &amp;&amp; c &lt;= 'F') return (byte)(c-'A'+10);
    return 0;
}
