#define AT_O8K_UART_DEVICE_NAME "/dev/ttyUSB0"

#define BUF_LEN 2048*2

static char buf[BUF_LEN];

static char buf_1[BUF_LEN];

static char buf_5[BUF_LEN];

static char buf_6[BUF_LEN];

static int fd1 = -1;

static int fd5 = -1;

static int fd6 = -1; 

static at_open_dev(char *dev_name)

{

    int fd = -1;

    struct termios newtio; 

    fd = open(dev_name, O_RDWR | O_NOCTTY | O_NONBLOCK);

    if (fd < 0)

    {

        return fd;

    }

    bzero(&newtio, sizeof(struct termios));     /* clear struct for new port settings */

    tcgetattr(fd, &newtio);

    /* step 1 : control mode flags */

    // No parity (8N1): 8bit word length, no parity, 1bit stop bit

    newtio.c_cflag = 0;         /* This is neccessary */

    newtio.c_cflag &= ~CSIZE;     

    newtio.c_cflag |= CS8;        /* uart word length : 8bit */

    newtio.c_cflag &= ~CSTOPB;      /* stop bit : 1 bit */

    newtio.c_cflag &= ~PARENB;      /* no parity */

    // 115200 baudrate : 

    newtio.c_cflag |= B115200;      /* 115200 baud rate */

    newtio.c_cflag |= CREAD;      /* care all characters => ignore all characters if CREAD is not set */

    // omap3530: enable or disable flow control both are ok

    /* If non-blocking mode is set, or CLOCAL mode is set,

     * we don’t want to wait for the modem status lines to

     * indicate that the port is ready.

     */

#if 1 /* 2010-07-08 : Always enable this,but need to disable this if hardware flow control is not enabled  */

    // special set when one uart is used as a modem connector : 

    printf("Hardware flow control enabled\n");

    newtio.c_cflag &= ~CLOCAL;    /* 不忽略MODEM控制线 */

    newtio.c_cflag |= HUPCL;      /* 在最后一个进程关闭设备后，降低MODEM控制线(挂断) */

    newtio.c_cflag |= CRTSCTS;    /* enable hardware flow control when the uart is used as a modem connector */ 

#else

    printf("Hardware flow control disabled\n");

#endif

    /* step 2 : input mode flags */

    newtio.c_iflag &= ~IGNPAR;      /* ignore Parity error and Frame error */ 

    newtio.c_iflag &= ~IGNBRK;      /* ignore Break interrupt */

    newtio.c_iflag &= ~IGNPAR;      /* ignore overrun error */

    /* step 3 : output mode flags */

    newtio.c_oflag &= ~OPOST;         /* raw output : 不要对输出做任何预处理 */ 

    /* step 4 : local mode flags */

    newtio.c_lflag &= ~(ICANON | ECHO | ECHOE | ISIG);  /* raw input */ 

    /* step 5 : line discipline */

    newtio.c_line = 0;          /* use line dicipline 0 */

#if 0

/* step 6 : control characters /     / set control chars */

newtio.c_cc[VINTR] = 3;   /* C-c */

newtio.c_cc[VQUIT] = 28;  /* C-\ */

newtio.c_cc[VERASE] = 127;  /* C-? */

newtio.c_cc[VKILL] = 21;  /* C-u */

newtio.c_cc[VEOF] = 4;    /* C-d */

newtio.c_cc[VSTART] = 17; /* C-q */

newtio.c_cc[VSTOP] = 19;  /* C-s */

newtio.c_cc[VSUSP] = 26;  /* C-z */ 

#endif  

    tcflush(fd, TCIFLUSH);        /* clear input buffer */

    tcsetattr(fd, TCSANOW, &newtio);

    return fd;

}

int main(int argc, char *argv[])

{

    int len;

    int iRet = 0;

    int i;

    char AtBuf[BUF_LEN];

    struct termios newtio;  

    assert(2 == argc);

    //**

    //* Open and configure channel 5

    //fd5 = at_open_dev(AT_O8K_UART_DEVICE_NAME);

    fd5 = at_open_dev(argv[1]);

    if(fd5>0)

    {

        //printf("Open %s success\n", AT_O8K_UART_DEVICE_NAME);

        printf("Open %s success\n", argv[1]);

    }

    while (1)

    {

        memset(AtBuf, 0, BUF_LEN);

        memset(buf, 0, BUF_LEN);

        memset(buf_1, 0, BUF_LEN);

        memset(buf_5, 0, BUF_LEN);

        memset(buf_6, 0, BUF_LEN);

        len = 0;    

        struct pollfd read_poll[2];

        read_poll[0].fd = fd5;      /* fd5 */

        read_poll[0].events = POLLIN;

        read_poll[0].revents = 0;

        read_poll[1].fd = 0;      /* 0 */

        read_poll[1].events = POLLIN;

        read_poll[1].revents = 0;

        //sleep(2);

        iRet = poll(read_poll, 4, -1);       

        if (read_poll[0].revents & POLLIN)        /* port 5 */

        {

            len = read(fd5, buf_5, BUF_LEN - 10);

            if (len <= 0)

            {

                printf("read nothing \n");

                continue;

            }

            buf_5[len] = 0;

            for (i = 0; i < len; i++)

                printf("%02x ", buf_5[i]);

            printf("\nThe read buf is: %s \n", buf_5);

        }

        if (read_poll[1].revents & POLLIN)        /* standard input */

        {

            len = 0;

            len = read(0, AtBuf, BUF_LEN - 10);

            //*****

            //  Because the standart input(tty) will add a "\n" after reading.So we must cut it.

            //  while scanf() will not add a "\n" after reading.

            AtBuf[–len] = 0;         

            if (len)

            {             

                sprintf(AtBuf, "%s\r", AtBuf);

                len = strlen(AtBuf);

                if (strncmp(AtBuf, "kke", 3) == 0)

                    goto kk_exit;

                usleep(100 * 1000);

                len = write(fd5, AtBuf, len);   

                memset(AtBuf, 0, BUF_LEN);  

                printf("\n");

                len = 0;

            }

        }

}

    kk_exit : close(fd1);

    close(fd5);

    close(fd6);   

    return 0;

}
